    X = average_housing_prices[['DATE_ordinal']]  # Features
    y = average_housing_prices['VALUE']  # Target variable

    model = LinearRegression()

    # Predicting the values for the line of best fit
    average_housing_prices['fit'] = model.predict(X)

    # Plotting the housing price data and line of best fit

# Time series decomposition
if not average_housing_prices.empty:
    # Ensure 'VALUE' is present in the DataFrame
    if 'VALUE' in average_housing_prices.columns:
        # Set REF_DATE as the index for decomposition
        average_housing_prices.set_index('REF_DATE', inplace=True)
        
        # Perform seasonal decomposition on the 'VALUE' column
        decomposition = seasonal_decompose(average_housing_prices['VALUE'], model='additive')

        # Plotting the decomposition
        plt.figure(figsize=(12, 10))

        plt.subplot(411)
        plt.plot(decomposition.observed, label='Observed')
        plt.title('Observed')
        plt.legend()

        plt.subplot(412)
        plt.plot(decomposition.trend, label='Trend', color='orange')
        plt.title('Trend')
        plt.legend()

        plt.subplot(413)
        plt.plot(decomposition.seasonal, label='Seasonal', color='green')
        plt.title('Seasonal')
        plt.legend()

        plt.subplot(414)
        plt.plot(decomposition.resid, label='Residual', color='red')
        plt.title('Residual')
        plt.legend()

        plt.tight_layout()
        plt.show()
    else:
        print("VALUE column not found in average_housing_prices.")
else:
    print("\nNo data available for plotting.")
